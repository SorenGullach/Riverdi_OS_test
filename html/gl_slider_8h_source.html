25template<int len, const FontItem *fontFamily>
26 class glSlider : public glWidgetLink, public glButtonTheme, private glPlot2DHelper
27 {
28 public:
29 glSlider(const glRegion_t &region, const char *text, const glButtonTheme &theme = glButtonTheme())
30 : glWidgetLink("Slider", region)
31 , glButtonTheme(theme)
32 , _Text(region.Inflate(-_BorderWidth), text, fontFamily, theme._TextColor)
33 {
34 Add(&_Text);
35 Horisontal = region.Height() < region.Width();
36 _CornerRadius = std::min(region.Height(), region.Width()) / 4;
37 CalcCursor(_Region.Inflate(-_BorderWidth));
38 }
39
40 protected:
41
42 void CalcCursor(const glRegion_t &r)
43 {
44 _InvalidatedRegion = _InvalidatedRegion.Union(Cursor); // old invalid region
45 if (Horisontal)
46 Cursor = glRegion_t(r.L() + Pos, r.T(), r.L() + 2*_CornerRadius + Pos, r.B());
47 else
48 Cursor = glRegion_t(r.L(), r.T() + Pos, r.R(), r.T() + 2*_CornerRadius + Pos);
49 _InvalidatedRegion = _InvalidatedRegion.Union(Cursor); // add new invalid region
50 _Text.Text("%d", Pos);
51 InvalidateChilds();
52 }
53
54 bool Touch(const glTouchPoint_t &point) override
55 {
56 bool Handled = false;
57 glRegion_t r = _Region.Inflate(-_BorderWidth);
58
59 if (_Region.IsInside(point.X, point.Y))
60 {
61 if (point.TipAction == glTouchPoint_t::eTipAction::Tip && !Pressed)
62 {
63 Pressed = true;
64 CalcCursor(r);
65 Handled = true;
66 }
67 if (point.TipAction == glTouchPoint_t::eTipAction::Up && Pressed)
68 {
69 Click = true;
70 Pressed = false;
71 CalcCursor(r);
72 Handled = true;
73 }
74 if (
75 (point.TipAction == glTouchPoint_t::eTipAction::Tip ||
76 point.TipAction == glTouchPoint_t::eTipAction::Slide) &&
77 Pressed)
78 {
79 if (Horisontal)
80 {
81 int p = point.X - r.L();
82 if (p != Pos && p > 0)
83 {
84 Pos = p;
85 if (Pos > r.Width() - 2*_CornerRadius)
86 Pos = r.Width() - 2*_CornerRadius;
87 CalcCursor(r);
88 Handled = true;
89 }
90 }
91 else
92 {
93 int p = point.Y - r.T();
94 if (p != Pos && p > 0)
95 {
96 Pos = p;
97 if (Pos > r.Height() - 2*_CornerRadius)
98 Pos = r.Height() - 2*_CornerRadius;
99 CalcCursor(r);
100 Handled = true;
101 }
102 }
103 }
104 }
105 else
106 {
107 if (Pressed)
108 {
109 Pressed = false;
110 CalcCursor(r);
111 Handled = true;
112 }
113 }
114 return Handled;
115 }
116
117 // redraw your self
118 virtual void Redraw() override
119 {
120 if (_InvalidatedRegion.IsEmpty()) return;
121
122 glRegion_t r = _Region.Inflate(-_BorderWidth + 1);
123 PlotRectangleRoundFill(r, _CornerRadius, _BackColor);
124
125 PlotRectangleRoundFill(Cursor, _CornerRadius, _PressedColor);
126
127 PlotBorder(_Region, _BorderWidth, _CornerRadius, _BorderColor);
128
129 _InvalidatedRegion.Empty();
130 }
131
132 glText<len> _Text;
133 bool Horizontal = true;
134 P_t Pos = 0;
135 glRegion_t Cursor;
136 bool Horisontal;
137
138 bool Click = false, Pressed = false;
139 };
