31class glVideoMemoryPlot : protected glVideoMemory
32{
33public:
34 // plot in invalid region and box
35 void Plot(const gl2DPoint_t &box, P_t x, P_t y, const glColor_t &color)
36 {
37 gl2DPoint_t IVR = InvalidRegion.Intersection(box);
38 if (IVR.IsInside(x, y))
39 {
40 assert(x < _ScreenWidth && y < _ScreenHeight);
41 pVM[x + y*_ScreenWidth] = color.operation == glColor_t::eColorOperation::Replace ? (glARGB_t)color : color.Color(pVM[x + y*_ScreenWidth]);
42 }
43 }
44
45 // plot in invalid region
46 void Plot(P_t x, P_t y, const glColor_t &color)
47 {
48 if (InvalidRegion.IsInside(x, y))
49 {
50 assert(x < _ScreenWidth && y < _ScreenHeight);
51 if (color.operation == glColor_t::eColorOperation::Replace)
52 pVM[x + y*_ScreenWidth] = (glARGB_t)color;
53 else
54 pVM[x + y*_ScreenWidth] = color.Color(pVM[x + y*_ScreenWidth]);
55 //pVM[x + y*_ScreenWidth] = color.operation == glColor_t::eColorOperation::Replace ? (glARGB_t)color : color.Color(pVM[x + y*_ScreenWidth]);
56 }
57 }
58
59 // plot in invalid region
60 void Plot(const glPoint_t &p, const glColor_t &color) { Plot(p.X, p.Y, color); }
61 // plot in invalid region and box
62 void Plot(const gl2DPoint_t box, const glPoint_t &p, const glColor_t &color) { Plot(box, p.X, p.Y, color); }
63};

66class glWidgetLink : public glLink
67{
68public:
69 glWidgetLink(char const *name, const gl2DPoint_t &region)
70 : _Name(name)
71 , _Region(region.Normalize()) {}
72
73 // Add a widget to the page
74 void Add(glWidgetLink *child)
75 {
76 ChainChilds.Add(child);
77 }
78
79 virtual const char * Name() { return _Name; }
80 virtual void Name(const char *name) { _Name = name; }
81
82 virtual gl2DPoint_t &Region() { return _Region; }
83 virtual void MoveTo(const gl2DPoint_t &region) { _Region = region; }
84
85 // redraw your self
86 virtual void Redraw() = 0;
87 // Called upon touch change
88 virtual bool Touch(const glTouchPoint_t &) = 0;
89 virtual gl2DPoint_t InvalidRegion()
90 {
91 return _Region;
92 }
93
94 // called down link to redraw
95 void UpdateLook()
96 {
97 Redraw(); // draw this widget
98
99 // Update Childs
100 if (ChainChilds.Head() != nullptr)
101 ChainChilds.Head()->UpdateLook();
102
103 // Update widgets
104 if (pNext != nullptr)
105 pNext->UpdateLook();
106 }
107
108 // called down link to update state
109 void UpdateState(glTouchPoint_t &point, bool &Handled)
110 {
111 // Update Childs
112 if (ChainChilds.Head() != nullptr && !Handled)
113 ChainChilds.Head()->UpdateState(point, Handled);
114
115 // Update widgets
116 if (pNext != nullptr && !Handled)
117 pNext->UpdateState(point, Handled);
118
119 if(!Handled)
120 Handled |= Touch(point);
121 }
122
123 void InvalidateMe()
124 {
125 //Printf("%s Invalidate\n", Name);
126 ImInvalidated = true; // this obj
127 if (ChainChilds.Head() != nullptr)
128 ChainChilds.Head()->InvalidateSiblings(); // & all childs
129 }
130
131 void InvalidateSiblings()
132 {
133 InvalidateMe();
134
135 // Update siblings
136 if (pNext != nullptr)
137 pNext->InvalidateSiblings();
138 }
139
140 bool IsInvalidated(gl2DPoint_t &invalidRegion)
141 {
142 bool Invalid = false;
143 if (ImInvalidated)
144 {
145 invalidRegion = invalidRegion.Union(InvalidRegion());
146 Invalid = true; // all childern is also invalid
147 }
148 else if (ChainChilds.Head() != nullptr)
149 {
150 // ask all childs
151 Invalid = ChainChilds.Head()->IsInvalidated(invalidRegion);
152 }
153
154 if (pNext != nullptr)
155 {
156 // ask all objs
157 Invalid |= pNext->IsInvalidated(invalidRegion);
158 }
159 return Invalid;
160 }
161
162 // buble event to top
163 void Event(glEvent event)
164 {
165 if (pPrev != nullptr)
166 pPrev->Event(event);
167 }
168
169protected:
170 char const *_Name = "glWidget";
171 // is widget invalidated
172 bool ImInvalidated = true;
173 gl2DPoint_t _Region;
174private:
175 glChain<glWidgetLink> ChainChilds;
176};
