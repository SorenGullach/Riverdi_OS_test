41 void Init(cbDisplay *cb)
42 {
43 cbClass = cb;
44 ResetDisplay();
45
46 TimBackLight.Prescaler(TimBackLight.FindPrescalerValue(5000'00)); // 1MHz
47 TimBackLight.Periode(999);
48 TimBackLight.Mode(1, Timer::eMode::PWM1); // set CCR1 to PWM1
49 TimBackLight.Compare(1, 200); // set CCR1 to xxx
50
51 hwLTDC::Init();
52
53 TimBackLight.Start();
54 }
55
56 void Intencity(uint16_t value)
57 {
58 if (value == 0)
59 {
60 TimBackLight.Stop();
61 return;
62 }
63 TimBackLight.Compare(1, value);
64 TimBackLight.Start();
65 }

67 void UnitTest(int testLevel)
68 {
69 Printf("UnitTest of Display\n");
70 for (int i = 0; i < 5; i++)
71 {
72 hwDSI::UnitTest(true);
73 osDelay(1000); // Delay for 1 second
74 }
75 hwDSI::UnitTest(false); // Turn off display test and enable LTDC
76
77 Layer(1, nullptr, 0, 0, hwLTDC::Width(), hwLTDC::Height());
78 Layer(2, nullptr, 0, 0, hwLTDC::Width(), hwLTDC::Height());
79
80 for (uint16_t i = 0; i < 1; i++)
81 {
82 for (uint16_t r = 0; r <= 255; r++)
83 {
84 hwLTDC::BackgroundColor(r, 0, 0);
85 osDelay(10);
86 }
87 for (uint16_t g = 0; g <= 255; g++)
88 {
89 hwLTDC::BackgroundColor(0, g, 0);
90 osDelay(10);
91 }
92 for (uint16_t b = 0; b <= 255; b++)
93 {
94 hwLTDC::BackgroundColor(0, 0, b);
95 osDelay(10);
96 }
97 }
98
99 for (uint16_t x = 0; x <= 255; x++)
100 {
101 glColor_t t = glColor_t::RainbowColorGradient(x,255);
102 hwLTDC::BackgroundColor(t.R, t.G, t.B);
103 osDelay(50);
104 }
105 osDelay(1000);
106 hwLTDC::BackgroundColor(125, 125, 125);
107
108 if (testLevel > 2)
109 UnitTest1();
110
111 Printf("UnitTest of Display end \n");
112 }

114 void UnitTest1()
115 {
116 Printf("UnitTest1 of Display\n");
117
118 // Define the number of lines for the display
119 uint32_t Height1 = hwLTDC::Height(); // layer 1 size
120 uint32_t Width1 = hwLTDC::Width();
121 const uint32_t Height2 = 100; // layer 2 size
122 uint32_t Width2 = hwLTDC::Width(600);
123 const uint32_t Y2 = (Height1 - Height2) / 2;
124 const uint32_t X2 = (Width1 - Width2) / 2;
125
126 //__RAM_ALIGNED(4) static glColor_t VM2[200*300+64];
127
128 uint32_t size = SDRAM_SIZE / 2; // Width2*Height2*sizeof(glColor_t);
129 assert(size > Width1*Height1*sizeof(glColor_t) && "SDRam too small");
130 glColor_t *pVM1 = (glColor_t *)SDRAM_START;
131 glColor_t *pVM2 = (glColor_t *)(SDRAM_START + size);
132 // glColor_t *pVM2 = (glColor_t *)&VM2;
133
134 Printf(" Display layer 1 x=%d,y=%d,w=%d H=%d Adr=0x%8X\n", 0, 0, Width1, Height1, (uint32_t)pVM1);
135 Printf(" Display layer 2 x=%d,y=%d,w=%d H=%d Adr=0x%8X\n", X2, Y2, Width2, Height2, (uint32_t)pVM2);
136
137 // back color for background
138 BackgroundColor(255, 0, 0);
139
140 // a rainbow band at "X"
141 glColor_t t = glColor_t::RainbowColorGradient(500, Width2);
142 DefaultColor(2, t.R, t.G, t.B, t.A);
143
144 Layer(1, pVM1, 0, 0, 0 + Width1, 0 + Height1);
145 Layer(2, pVM2, X2, Y2, X2 + Width2, Y2 + Height2);
146
147 BlendingFactors(1, eBlendingFactor1::F1_CA, eBlendingFactor2::F2_CA, 100);
148 BlendingFactors(2, eBlendingFactor1::F1_CA, eBlendingFactor2::F2_CA, 100);
149
150#define TDELAY 100
151 // Update pixel data for display
152 uint32_t x = 0, y = 0;
153 pVM1[x + y*Width1].A = 255;
154 pVM1[x + y*Width1].R = 255;
155 pVM1[x + y*Width1].G = 0;
156 pVM1[x + y*Width1].B = 0;
157 x = Width1 - 1, y = 0;
158 pVM1[x + y*Width1].A = 255;
159 pVM1[x + y*Width1].R = 0;
160 pVM1[x + y*Width1].G = 255;
161 pVM1[x + y*Width1].B = 0;
162 for (uint32_t y = 0; y < Height1; y++)
163 {
164 uint32_t x = Width1 - y - 1;
165 pVM1[x + y*Width1].A = 255; // cross line
166 pVM1[x + y*Width1].R = 255;
167 pVM1[x + y*Width1].G = 255;
168 pVM1[x + y*Width1].B = 255;
169 }
170 osDelay(1000);
171 for (uint32_t y = 0; y < Height1; y++)
172 {
173 uint32_t x = y;
174 pVM1[x + y*Width1].A = 255; // cross line
175 pVM1[x + y*Width1].R = 255;
176 pVM1[x + y*Width1].G = 255;
177 pVM1[x + y*Width1].B = 255;
178 }
179 osDelay(TDELAY);
180
181 for (uint32_t y = 0; y < Height1; y++)
182 {
183 uint32_t x = 0;
184 pVM1[x + y*Width1].A = 255; // right line
185 pVM1[x + y*Width1].R = 0;
186 pVM1[x + y*Width1].G = 0;
187 pVM1[x + y*Width1].B = 255;
188 }
189 osDelay(TDELAY);
190 for (uint32_t y = 0; y < Height1; y++)
191 {
192 uint32_t x = Width1 - 1;
193 pVM1[x + y*Width1].A = 255; // left line
194 pVM1[x + y*Width1].R = 255;
195 pVM1[x + y*Width1].G = 0;
196 pVM1[x + y*Width1].B = 0;
197 }
198 osDelay(TDELAY);
199
200 for (uint32_t x = 0; x < Width1; x++)
201 {
202 uint32_t y = 0;
203 pVM1[x + y*Width1].A = 255; // top line
204 pVM1[x + y*Width1].R = 255;
205 pVM1[x + y*Width1].G = 0;
206 pVM1[x + y*Width1].B = 0;
207 }
208 osDelay(TDELAY);
209 for (uint32_t x = 0; x < Width1; x++)
210 {
211 uint32_t y = Height1 - 1;
212 pVM1[x + y*Width1].A = 255; // bottom line
213 pVM1[x + y*Width1].R = 0;
214 pVM1[x + y*Width1].G = 255;
215 pVM1[x + y*Width1].B = 0;
216 }
217 osDelay(TDELAY);
218
219 for (uint32_t i = 0; i < 2; i++)
220 for (uint32_t x = 0; x < Width2; x++)
221 {
222 // Calculate the color value
223 glColor_t t = glColor_t::RainbowColorGradient(x,Width2);
224 for (uint32_t y = 0; y < Height2; y++)
225 {
226 pVM2[x + y*Width2] = t;
227 }
228 }
229 osDelay(TDELAY);
230
231 Printf("UnitTest1 of Display end \n");
232 }
