49 // Calculate the total width of the text based on the font information
50 P_t TextWidth()
51 {
52 P_t width = 0; // Initialize the width counter
53 int index = 0; // Initialize the character index
54 char character;
55
56 // Iterate over each character in the text until the null terminator is encountered
57 while ((character = _Text[index]) != '\0')
58 {
59 // Retrieve font information for the character
60 const FontItem* font = &FontFamily[character - FontFamily->c]; // Adjust font pointer based on character
61
62 // Ensure the retrieved font corresponds to the current character
63 assert(font->c == character);
64
65 // Accumulate the width of the character
66 width += font->w;
67
68 // Move to the next character in the text
69 index++;
70 }
71
72 return width; // Return the total width of the text
73 }

80 virtual void Redraw() override
81 {
82 // Check if the widget needs to be redrawn

85 // Calculate the center position of the widget
86 gl2DPoint_t pos = Center();

88 glRectangle(_Position, glColors::BLACK).Draw();

91 // Initialize variables
92 glColor_t color = _Color;
93 uint8_t pix[200];
94 P_t xs = pos.L();
95 char *pc = _Text;
96
97 // Iterate over each character in the text
98 while (*pc != 0)
99 {
100 // Retrieve font information for the character
101 const FontItem *f = &FontFamily[*pc++ -FontFamily->c];
102 assert(f->w < sizeof(pix)); // Ensure buffer size is not exceeded
103 P_t ys = pos.T() + f->yb;
104
105// glRectangle(f->xb, pos.T() + f->yb, f->xe, pos.T()f->ye, glColors::GREEN).Draw();

107 // Iterate over each row of the character
108 for (int l = 0, y = ys; l < f->StrideY/* && y < pos.B()*/; ++l, ++y)
109 {
110 // Extract pixel data for the row
111 CharPix(f, l, pix, _Color.A);
112
113 // Iterate over each pixel in the row
114 for (int x = 0; x < f->w; ++x)
115 {
116 // Plot the pixel if its value is greater than 0
117 if (pix[x] > 0)
118 {
119 color.A = pix[x];
120 Plot(xs + x, y, color);
121 }
122 }
123 }
124
125 // Move the horizontal position for the next character
126 xs += f->w;
127 }
128

138 // Function to extract pixel data for a character
139 void CharPix(const FontItem *f, int l, uint8_t pix[], C_t alpha)
140 {
141 // Initialize variables
142 int BitsPerPixel = f->BitsPerPixel;
143 uint8_t *Data = &f->Data[f->StrideX * l]; // Pointer to the start of character data
144
145 // Clear the pixel buffer
146 memset(pix, 0, f->w);
147
148 // Variables for bit manipulation
149 int BPP = 1 << f->BitsPerPixel;
150 int bitPosition = 0;
151 uint8_t currentByte = 0;
152 int pixelIndex = 0; // Index for accessing elements in the pixel buffer
153
154 // Iterate over each pixel in the character
155 for (int x = 0; x < f->w && pixelIndex < f->w; ++x)
156 {
157 // Unpack each pixel from the packed data
158 //for (int b = BitsPerPixel - 1; b >= 0 && pixelIndex < f->w; --b)
159 for (int b = 0; b < BitsPerPixel && pixelIndex < f->w; ++b)
160 {
161 // Check if we need to read a new byte
162 if (bitPosition % 8 == 0)
163 currentByte = Data[bitPosition / 8];
164
165 // Extract the pixel value from the current byte
166 uint8_t pixelValue = (currentByte >> (8 - (bitPosition % 8) - BitsPerPixel)) & ((1 << BitsPerPixel) - 1);
167 if (pixelValue > 0)
168 pix[pixelIndex] = alpha / (BPP - pixelValue);
169
170 // Increment the bit position and pixel index
171 bitPosition += BitsPerPixel;
172 ++pixelIndex;
173 }
174 }
175 }

178 // find center pos
179 gl2DPoint_t Center()
180 {
181 // Initialize variables
182 P_t w = TextWidth();
183 P_t h = TextHeight();
184
185 // Calculate x-coordinate for centering
186 P_t x = _Position.L();
187 if (w < _Position.Width())
188 x += (_Position.Width() - w) / 2;
189
190// Calculate y-coordinate for centering
191 P_t y = _Position.T();
192 if (h < _Position.Height())
193 y += (_Position.Height() - h) / 2;
194
195// Return intersection point between position and calculated rectangle
196 return _Position.Intersection(gl2DPoint_t(x, y, x + w, y + h));
197 }
