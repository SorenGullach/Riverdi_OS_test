9/*
10 The Memory Protection Unit (MPU) divides the memory map into a number of regions, and
11 defines the location, size, access permissions, and memory attributes of each region. It
12 supports:
13 • Independent attribute settings for each region.
14 • Overlapping regions.
15 • Export of memory attributes to the system.
16 The memory attributes affect the behavior of memory accesses to the region. The Cortex®-
17 M7 MPU defines:
18 • 8 or 16 separate memory regions, 0-7 or 0-15.
19 • A background region.
20 When memory regions overlap, a memory access is affected by the attributes of the region
21 with the highest number. For example, the attributes for region 7 take precedence over the
22 attributes of any region that overlaps region 7.
23 The background region has the same memory access attributes as the default memory
24 map, but is accessible from privileged software only.
25 The Cortex®-M7 MPU memory map is unified. This means instruction accesses and data
26 accesses have same region settings.
27 If a program accesses a memory location that is prohibited by the MPU, the processor
28 generates a MemManage fault. This causes a fault exception, and might cause termination
29 of the process in an OS environment. In an OS environment, the kernel can update the MPU
30 region setting dynamically based on the process to be executed. Typically, an embedded
31 OS uses the MPU for memory protection.
32 **/

139 (uint32_t)accessPermission << MPU_RASR_AP_Pos |

143 (uint32_t)bufferable << MPU_RASR_B_Pos |
144 (uint32_t)size << MPU_RASR_SIZE_Pos |

148 // Enable MPU

152 enum class eControlMode_Hartfault
153 {
154 /*
155 Enables the operation of MPU during hard fault, NMI, and FAULTMASK
156 handlers.
157 When the MPU is enabled:
158 0: MPU is disabled during hard fault, NMI, and FAULTMASK handlers,
159 regardless of the value of the ENABLE bit.
160 1: The MPU is enabled during hard fault, NMI, and FAULTMASK handlers.
161 When the MPU is disabled, if this bit is set to 1 the behavior is
162 Unpredictable.
163 */
164 None = 0,
165 HartFault_NMI = 1
166 };
167 enum class eControlMode_Privileged
168 {
169 /*
170 Enables privileged software access to the default memory map:
171 0: If the MPU is enabled, disables use of the default memory map. Any
172 memory access to a location not covered by any enabled region causes a
173 fault.
174 1: If the MPU is enabled, enables use of the default memory map as a
175 background region for privileged software accesses.
176 When enabled, the background region acts as if it is region number -1. Any
177 region that is defined and enabled has priority over this default map.
178 If the MPU is disabled, the processor ignores this bit.
179 */
180 None = 0,
181 Privileged_default = 1
182 };
183 // Enable the MPU
184 static void Enable(eControlMode_Hartfault hartfault, eControlMode_Privileged privileged)
185 {
186 MPU->CTRL =
187 (uint32_t)hartfault << MPU_CTRL_HFNMIENA_Pos |
188 (uint32_t)privileged << MPU_CTRL_PRIVDEFENA_Pos |
189 MPU_CTRL_ENABLE_Msk;
190 // Enable fault exceptions
191 SCB->SHCSR |= SCB_SHCSR_MEMFAULTENA_Msk;
192
193 // Ensure MPU setting take effects
194 __DSB();
195 __ISB();
196 }
197 // Disable MPU
198 static void Disable()
199 {
200 // Make sure outstanding transfers are done
201 __DMB();
202 // Disable fault exceptions
203 SCB->SHCSR &= ~SCB_SHCSR_MEMFAULTENA_Msk;
204 // Disable the MPU and clear the control register
205 MPU->CTRL = 0;
206 }
