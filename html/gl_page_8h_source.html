113 virtual bool Touch(const glTouchPoint_t &point) final
114 {
115 if (InSlide)
116 {
117 if (point.TipAction == glTouchPoint_t::eTipAction::Up)
118 InSlide = false;
119// Printf("%s %s\n", _Name, InSlide ? "InSlide" : "");
120// return true;
121 }
122 if (point.SlideAction == glTouchPoint_t::eSlideAction::Right &&
123 EventAction.U.Slide.Action != glTouchPoint_t::Right)
124 {
125 InSlide = true;
126// Printf("%s slide right\n", _Name);
127 EventAction.EventType = glEvent::eEventType::Slide;
128 EventAction.U.Slide.Action = glTouchPoint_t::Right;
129 Event(EventAction);
130 // return true;
131 }
132 else if (point.SlideAction == glTouchPoint_t::eSlideAction::Left &&
133 EventAction.U.Slide.Action != glTouchPoint_t::Left)
134 {
135 InSlide = true;
136// Printf("%s slide left\n", _Name);
137 EventAction.EventType = glEvent::eEventType::Slide;
138 EventAction.U.Slide.Action = glTouchPoint_t::Left;
139 Event(EventAction);
140 // return true;
141 }
142 return false;
143 }
