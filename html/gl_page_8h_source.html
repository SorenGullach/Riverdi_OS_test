98 virtual void Touch(const glTouchPoint_t &point) final
99 {
100 if (InSlide)
101 {

103 InSlide = false;
104 Printf("%s %s\n", _Name, InSlide ? "InSlide" : "");
105 return;
106 }
107 if (point.SlideAction == glTouchPoint_t::eSlideAction::Right &&
108 EventAction.U.Slide.Action != glTouchPoint_t::Right)
109 {
110 InSlide = true;
111 Printf("%s slide right\n", _Name);
112 EventAction.EventType = glEvent::eEventType::Slide;
113 EventAction.U.Slide.Action = glTouchPoint_t::Right;
114 Event(EventAction);
115 }
116 else if (point.SlideAction == glTouchPoint_t::eSlideAction::Left &&
117 EventAction.U.Slide.Action != glTouchPoint_t::Left)
118 {
119 InSlide = true;
120 Printf("%s slide left\n", _Name);
121 EventAction.EventType = glEvent::eEventType::Slide;
122 EventAction.U.Slide.Action = glTouchPoint_t::Left;
123 Event(EventAction);
124 }
125 }

129 Printf("%s Event\n", _Name);
