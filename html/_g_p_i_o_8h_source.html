152 void SetMode(R_t<uint8_t, 0, 15>af)
153 {
154 pGPIO->MODER &= ~(0b11UL << 2*Pin); // 00
155 pGPIO->MODER |= 0b10UL << 2*Pin; // AF
156 if (Pin < 8)
157 {
158 pGPIO->AFR[0] &= ~(0xFUL << 4*Pin);
159 pGPIO->AFR[0] |= (uint32_t)af << 4*Pin;
160 }
161 else
162 {
163 pGPIO->AFR[1] &= ~(0xFUL << 4*(Pin - 8));
164 pGPIO->AFR[1] |= (uint32_t)af << 4*(Pin - 8);
165 }
166 }

168 void SetMode(eMode mode)
169 {
170 pGPIO->MODER &= ~(0b11UL << 2*Pin); // 00
171 if (((uint32_t)mode & 0B1100) != 0) // interrupt
172 {
173 RCC->APB4ENR |= RCC_APB4ENR_SYSCFGEN;
174
175 volatile uint32_t *pEXTICR;
176 if (/*Pin >= 0 && */Pin <= 3)
177 pEXTICR = &(SYSCFG->EXTICR[0]);
178 else if (Pin >= 4 && Pin <= 7)
179 pEXTICR = &(SYSCFG->EXTICR[1]);
180 else if (Pin >= 8 && Pin <= 11)
181 pEXTICR = &(SYSCFG->EXTICR[2]);
182 else if (Pin >= 12 && Pin <= 15)
183 pEXTICR = &(SYSCFG->EXTICR[3]);
184 else
185 {
186 assert(false && "More code");
187 }
188
189 if (pGPIO == GPIOA) {}
190 if (pGPIO == GPIOB) {}
191 if (pGPIO == GPIOC) {}
192 if (pGPIO == GPIOD) {}
193 if (pGPIO == GPIOE) {}
194 if (pGPIO == GPIOF) {}
195 if (pGPIO == GPIOG) {}
196 if (pGPIO == GPIOH)
197 {
198 *pEXTICR &= ~(0B1111 << (Pin * 4));
199 *pEXTICR |= (0B0111 << (Pin * 4));
200 }
201 if (pGPIO == GPIOI) {}
202 if (pGPIO == GPIOJ) {}
203#ifdef GPIOK
204 if (pGPIO == GPIOK) {}
205#endif // GPIOK
206 else
207 assert("Wrong port");
208
209 uint32_t Falling = ((uint32_t)mode & (uint32_t)eMode::Int_Falling) ? 1:0;
210 uint32_t Rising = ((uint32_t)mode & (uint32_t)eMode::Int_Rising) ? 1 : 0;
211 if (/*Pin >= 0 && */Pin <= 21)
212 {
213 EXTI->RTSR1 &= ~(1U << Pin);
214 EXTI->RTSR1 |= (Rising << Pin);
215 EXTI->FTSR1 &= ~(1U << Pin);
216 EXTI->FTSR1 |= (Falling << Pin);
217 }
218 else if (Pin == 49)
219 {
220 EXTI->RTSR2 &= ~(1U << 17);
221 EXTI->RTSR2 |= (Rising << 17);
222 EXTI->FTSR2 &= ~(1U << 17);
223 EXTI->FTSR2 |= (Falling << 17);
224 }
225 else if (Pin == 51)
226 {
227 EXTI->RTSR2 &= ~(1U << 19);
228 EXTI->RTSR2 |= (Rising << 19);
229 EXTI->FTSR2 &= ~(1U << 19);
230 EXTI->FTSR2 |= (Falling << 19);
231 }
232 else if (Pin == 82)
233 {
234 EXTI->RTSR3 &= ~(1U << 19);
235 EXTI->RTSR3 |= (Rising << 19);
236 EXTI->FTSR3 &= ~(1U << 19);
237 EXTI->FTSR3 |= (Falling << 19);
238 }
239 else if (Pin >= 84 && Pin <= 86)
240 {
241 EXTI->RTSR3 &= ~(1U << (20 + (Pin - 84)));
242 EXTI->RTSR3 |= (Rising << (20 + (Pin - 84)));
243 EXTI->FTSR3 &= ~(1U << (20 + (Pin - 84)));
244 EXTI->FTSR3 |= (Falling << (20 + (Pin - 84)));
245 }
246 else
247 {
248 assert(false && "More code");
249 }
250
251 }
252 pGPIO->MODER |= (((uint32_t)mode) & 0B0011) << 2*Pin;
253 }

255 {

257 pGPIO->OTYPER |= 1UL << Pin;
258 else
259 pGPIO->OTYPER &= ~(1UL << Pin);
260 }
