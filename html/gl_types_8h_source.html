302 glRegion_t Union(const glRegion_t &region) const
303 {
304 if (IsEmpty()) return region;
305 if (region.IsEmpty()) return *this;
306
307 glRegion_t Region2 = region.Normalize();
308 glRegion_t ThisNorm = this->Normalize();
309
310 P_t newLeft = std::min(ThisNorm.L(), Region2.L());
311 P_t newTop = std::min(ThisNorm.T(), Region2.T());
312 P_t newRight = std::max(ThisNorm.R(), Region2.R());
313 P_t newBottom = std::max(ThisNorm.B(), Region2.B());

323 glRegion_t Intersection(const glRegion_t &region) const
324 {
325 if (IsEmpty() || region.IsEmpty()) return glRegion_t();
326
327 glRegion_t Region2 = region.Normalize();
328 glRegion_t ThisNorm = this->Normalize();
329
330 P_t newLeft = std::max(ThisNorm.L(), Region2.L());
331 P_t newTop = std::max(ThisNorm.T(), Region2.T());
332 P_t newRight = std::min(ThisNorm.R(), Region2.R());
333 P_t newBottom = std::min(ThisNorm.B(), Region2.B());
334
335 if (newLeft < newRight && newTop < newBottom)
336 {
337 return glRegion_t(newLeft, newTop, newRight, newBottom);
338 }
339 else
340 {
341 return glRegion_t();
342 }
