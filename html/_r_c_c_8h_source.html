169 /*
170 The RCC features three PLLs :
171 • A main PLL, PLL1, which is generally used to provide clocks to the CPUs and to some peripherals.
172 • Two dedicated PLLs, PLL2 and PLL3, which are used to generate the kernel clock for peripherals.
173 */

180 // Dumps the current settings

185 bool IsEnabled()
186 {
187 uint32_t on = (((Idx == 1) * RCC_CR_PLL1ON) | ((Idx == 2) * RCC_CR_PLL2ON) | ((Idx == 3) * RCC_CR_PLL3ON));
188 return (RCC->CR & on) != 0;
189 }
190 void Enable()
191 {
192 uint32_t on = (((Idx == 1) * RCC_CR_PLL1ON) | ((Idx == 2) * RCC_CR_PLL2ON) | ((Idx == 3) * RCC_CR_PLL3ON));
193 RCC->CR |= on; // Enable PLL
194 while (!IsEnabled()) __asm("");
195 }
