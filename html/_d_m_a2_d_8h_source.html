42 void Init() {
43 // Enable the DMA2D clock
44 RCC->AHB3ENR |= RCC_AHB3ENR_DMA2DEN;
45
46 // Reset the DMA2D peripheral
47 RCC->AHB3RSTR |= RCC_AHB3RSTR_DMA2DRST;
48 RCC->AHB3RSTR &= ~RCC_AHB3RSTR_DMA2DRST;
49
50 // Set the DMA2D mode to memory-to-memory
51 DMA2D->CR = 0; // Mode 0 is memory-to-memory
52
53 // Set the color mode to ARGB8888
54 DMA2D->FGPFCCR = 0x00; // fg color mode ARGB8888
55 DMA2D->BGPFCCR = 0x00; // bg color mode ARGB8888
56 DMA2D->OPFCCR = 0x00; // output color mode ARGB8888
57 }

68 void StartM2MTransfer(void* dstAddress, uint32_t width, uint32_t height, void* srcAddress, uint32_t lineoffset) {
69 // Set the DMA2D mode to memory-to-memory
70 DMA2D->CR = 0; // Mode 0 is memory-to-memory
71
72 DMA2D->AMTCR = (1 << 0) | (100 << 8); // Enable and set dead-time
73
74 // Configure source and destination addresses
75 DMA2D->FGMAR = (int32_t)srcAddress;
76 DMA2D->FGOR = lineoffset;
77 DMA2D->OMAR = (int32_t)dstAddress;
78 DMA2D->OOR = lineoffset;
79
80 // Set the number of lines and pixels per line
81 DMA2D->NLR = (height << DMA2D_NLR_NL_Pos) | (width << DMA2D_NLR_PL_Pos);
82
83 // Start the transfer
84 DMA2D->CR |= DMA2D_CR_START;
85 }

92 void StopTransfer() {
93 // Abort the transfer
94 DMA2D->CR |= DMA2D_CR_ABORT;
95 }
