74public:
75 void cbTouch(glTouchPoint_t &point) override
76 {
77 if (pCurrentPage == nullptr) return;
78 pCurrentPage->UpdateState(point);
79
80 switch (pCurrentPage->EventAction.EventType)
81 {
82 case glEvent::eEventType::Click:
83 break;
84 case glEvent::eEventType::Slide:
85 if (pCurrentPage->EventAction.U.Slide.Action == glTouchPoint_t::Right && pCurrentPage->Prev() != nullptr)
86 {
87 ((glPage *)pCurrentPage->Prev())->Invalidate();
88 pCurrentPage->EventAction.U.Slide.Action = glTouchPoint_t::None;
89 pCurrentPage = (glPage *)(pCurrentPage->Prev());
90 //Printf("%s Currentpage\n", pCurrentPage->Name);
91 }
92 if (pCurrentPage->EventAction.U.Slide.Action == glTouchPoint_t::Left && pCurrentPage->Next() != nullptr)
93 {
94 ((glPage *)pCurrentPage->Next())->Invalidate();
95 pCurrentPage->EventAction.U.Slide.Action = glTouchPoint_t::None;
96 pCurrentPage = (glPage *)(pCurrentPage->Next());
97 //Printf("%s Currentpage\n", pCurrentPage->Name);
98 }
99 break;
100 case glEvent::eEventType::None:
101 break;
102 }
103 pCurrentPage->EventAction.EventType = glEvent::eEventType::None;
104 }
105
106 void cbBlanking() override
107 {
108 if (pCurrentPage == nullptr) return;
109
110 if (!pCurrentPage->IsInvalidated(glVideoMemory::InvalidRegion)) // returns an invalidated region
111 return;
112
113 glVideoMemory::LimitRegion();
114 pCurrentPage->UpdateLook();
115 glVideoMemory::NextRegion();
116 }
117};
