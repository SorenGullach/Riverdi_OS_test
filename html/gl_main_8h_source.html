79 void cbTouch(const glTouchPoint_t &point) override
80 {
81
82 if (pCurrentPage == nullptr) return;
83 bool Handled = false;
84 pCurrentPage->UpdateState(point, Handled);
85
86 switch (pCurrentPage->EventAction.EventType)
87 {
88 case glEvent::eEventType::Slide:
89 if (pCurrentPage->EventAction.U.Slide.Action == glTouchPoint_t::Right && pCurrentPage->Prev() != nullptr)
90 {
91 ((glPageLink *)pCurrentPage->Prev())->InvalidateChilds();
92 pCurrentPage->EventAction.U.Slide.Action = glTouchPoint_t::None;
93 pCurrentPage = (glPageLink *)(pCurrentPage->Prev());
94 //Printf("%s Currentpage\n", pCurrentPage->Name);
95 }
96 if (pCurrentPage->EventAction.U.Slide.Action == glTouchPoint_t::Left && pCurrentPage->Next() != nullptr)
97 {
98 ((glPageLink *)pCurrentPage->Next())->InvalidateChilds();
99 pCurrentPage->EventAction.U.Slide.Action = glTouchPoint_t::None;
100 pCurrentPage = (glPageLink *)(pCurrentPage->Next());
101 //Printf("%s Currentpage\n", pCurrentPage->Name);
102 }
103 break;
104 case glEvent::eEventType::Click:
105 case glEvent::eEventType::None:
106 break;
107 }
108 pCurrentPage->EventAction.EventType = glEvent::eEventType::None;
109 }
110

127
128 pCurrentPage->UpdateLook();
129
130 glVideoMemory::CopyVideoMem();
131
132 // Printf(":\n");
133 }
