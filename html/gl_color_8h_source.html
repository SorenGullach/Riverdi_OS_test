322 static glColors FindNearestColor(const glColor_t &color)
323 {
324 glColors nearestColor = glColors::BLACK; // Default to black
325 float minDistance = std::numeric_limits<float>::max();
326
327 // Iterate through all colors in the enum list
328 for (uint32_t i = 0; i < glColorTableSize; ++i) {

333 float distance = sqrtf(powf(colorCurrent.R - color.R, 2) + powf(colorCurrent.G - color.G, 2) + powf(colorCurrent.B - color.B, 2));
